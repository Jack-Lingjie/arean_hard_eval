function changeStep(btn) {
let index = 0;
const active = document.querySelector(“.active”);
index = steps.indexOf(active);
steps[index].classList.remove(“active”);
if (btn === “next”) {
index++;
} else if (btn === “prev”) {
index–;
}
steps[index].classList.add(“active”);
}
